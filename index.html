char — Функсияҳо

char c = 'A';
ФунксияВазифаНатиҷа
char.IsLetter(c)Ҳарф аст?true
char.IsDigit(c)Рақам аст?false
char.IsWhiteSpace(c)Фосила аст?false
char.IsUpper(c)Калон аст?true
char.IsLower(c)Хурд аст?false
char.ToUpper(c)Калон мекунад'A'
char.ToLower(c)Хурд мекунад'a'

string — Функсияҳо

ФунксияВазифаНатиҷа
s.LengthДарозӣ11
s.ToUpper()Калон мекунад"HELLO WORLD"
s.ToLower()Хурд мекунад"hello world"
s.Trim()Фосилаҳои канорро мезанад"Hello World"
s.Replace("o","0")Иваз мекунад"Hell0 W0rld"
s.Contains("World")Дорад?true
s.StartsWith("He")Аз ин сар мешавад?true
s.EndsWith("ld")Ба ин тамом мешавад?true
s.Substring(0, 5)Қисме мегирад"Hello"
s.Split(' ')Ҷудо мекунад["Hello","World"]
s.IndexOf("W")Мавқеъ6
s.ToCharArray()Ба char[]['H','e','l','l','o'...]
